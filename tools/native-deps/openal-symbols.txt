# OpenAL function symbols exported by emscripten's library_openal.js.
# Pair with liblwjgl_openal_stubs.a (which adds alcGetProcAddress /
# alGetProcAddress that emscripten does not implement) when feeding
# gen-static-libs.py:
#
#   python3 gen-static-libs.py \
#       --symbol-list openal:openal-symbols.txt \
#       "openal:liblwjgl_openal_stubs.a"
#
# Signatures are spelled with plain C types (no ALC*/AL* typedefs) so this
# file is consumable without dragging in the AL/ALC headers. They MUST match
# the wasm-level signatures emscripten registers in the function table —
# call_indirect dispatches on signature, so a K&R `extern void foo();` traps
# when invoked through the (void*) entries gen-static-libs.py emits.
#
# Type mapping (per OpenAL 1.1 spec):
#   ALCdevice* / ALCcontext* / ALvoid*  → void*
#   ALCboolean / ALCenum / ALCint / ALCsizei
#     ALboolean / ALenum / ALint / ALsizei → int
#   ALCuint / ALuint                    → unsigned int
#   ALCchar / ALchar                    → char
#   ALfloat                             → float
#   ALdouble                            → double
#
# Regenerate the symbol list (names only) from the active emsdk with:
#
#   grep -oE '^\s+(al[A-Za-z][A-Za-z0-9_]*|alc[A-Z][A-Za-z0-9_]*|emscripten_al[A-Za-z0-9_]*)\s*:\s*\(' \
#       "$EMSDK/upstream/emscripten/src/library_openal.js" \
#     | sed -E 's/^[[:space:]]+//; s/[[:space:]]*:.*$//' \
#     | grep -v '__' | sort -u
#
# Confirmed against emscripten 3.1.56.
